697  template<typename T, typename... Args>
698  std::shared_ptr<T> Inject(Args&&... args) {
699  // Creator proxy, knows how to create the type we intend to inject
700  typedef autowiring::CreationRules<T, Args...> CreationRules;
701 
702  // Add this type to the TypeRegistry, also ensure that we initialize support for blind
703  // fast pointer cast to CoreObject.
704  (void) RegType<T>::r;
705  (void) autowiring::fast_pointer_cast_initializer<CoreObject, T>::sc_init;
706 
707  // First see if the base object type has already been injected. This is also necessary to
708  // ensure that a memo slot is created for the type by itself, in cases where the injected
709  // member does not inherit CoreObject and this member is eventually satisfied by one that does.
710  {
711  std::shared_ptr<T> pure;
712  FindByType(pure, true); //skip non-local slots.
713  if (pure)
714  return pure;
715  }
716 
717  // We must make ourselves current for the remainder of this call:
718  CurrentContextPusher pshr(shared_from_this());
719  std::shared_ptr<typename CreationRules::TActual> retVal(
720  CreationRules::New(*this, std::forward<Args>(args)...)
721  );
722 
723  // AutoInit if sensible to do so:
724  CallAutoInit(*retVal, has_autoinit<T>());
725 
726  try {
727  // Pass control to the insertion routine, which will handle injection from this point:
728  AddInternal(CoreObjectDescriptor(retVal, (T*)nullptr));
729  }
730  catch(autowiring_error&) {
731  // We know why this exception occurred. It's because, while we were constructing our
732  // type, someone else was constructing the same type at the same time. As a consequence,
733  // we will simply eat this exception, and handle it silently by returning the type that
734  // someone else has already attempted to construct, as per the documented behavior of
735  // Construct.
736  FindByType(retVal);
737  }
738 
739  // Factory registration if sensible to do so, but only after the underlying type has been added
740  // This ensures that any creation operations that happen as a consequence of factory registration
741  // can correctly back-reference the factory proper via autowiring
742  RegisterFactory(*retVal, autowiring::member_new_type<typename CreationRules::TActual>());
743  return std::static_pointer_cast<T>(retVal);
744  }
