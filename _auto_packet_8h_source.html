349  void DecorateImmediate(const T& immed, const Ts&... immeds) {
350  // These are the things we're going to be working with while we perform immediate decoration:
351  static const std::type_info* sc_typeInfo [] = {&typeid(T), &typeid(Ts)...};
352  const void* pvImmeds [] = {&immed, &immeds...};
353  DecorationDisposition* pTypeSubs[1 + sizeof...(Ts)];
354 
355  // None of the inputs may be shared pointers--if any of the inputs are shared pointers, they must be attached
356  // to this packet via Decorate, or else dereferenced and used that way.
357  static_assert(
358  !is_shared_ptr<T>::value &&
359  !is_any<is_shared_ptr<Ts>...>::value,
360  "DecorateImmediate must not be used to attach a shared pointer, use Decorate on such a decoration instead"
361  );
362 
363  // Perform standard decoration with a short initialization:
364  {
365  std::lock_guard<std::mutex> lk(m_lock);
366  for(size_t i = 0; i <= sizeof...(Ts); i++) {
367  pTypeSubs[i] = &m_decorations[*sc_typeInfo[i]];
368  if(pTypeSubs[i]->wasCheckedOut) {
369  std::stringstream ss;
370  ss << "Cannot perform immediate decoration with type " << sc_typeInfo[i]->name()
371  << ", the requested decoration already exists";
372  throw std::runtime_error(ss.str());
373  }
374 
375  // Mark the entry as appropriate:
376  pTypeSubs[i]->satisfied = true;
377  pTypeSubs[i]->wasCheckedOut = true;
378 
379  pTypeSubs[i]->m_pImmediate = pvImmeds[i];
380  }
381  }
382 
383  // Pulse satisfaction:
384  MakeAtExit([this, &pTypeSubs] {
385  // Mark entries as unsatisfiable:
386  for(auto pEntry : pTypeSubs) {
387  pEntry->satisfied = false;
388  pEntry->m_pImmediate = nullptr;
389  }
390 
391  // Now trigger a rescan to hit any deferred, unsatisfiable entries:
392  static const std::type_info* lamda_typeInfos [] = {&typeid(T), &typeid(Ts)...};
393  for(auto ti : lamda_typeInfos)
394  MarkUnsatisfiable(*ti);
395  }),
396  PulseSatisfaction(pTypeSubs, 1 + sizeof...(Ts));
397  }
