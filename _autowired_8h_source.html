210  template<class Fn>
211  void NotifyWhenAutowired(Fn fn) {
212  // Trivial initial check:
213  if(*this) {
214  fn();
215  return;
216  }
217 
218  // We pass a null shared_ptr, because we do not want this slot to attempt any kind of unregistration when
219  // it goes out of scope. Instead, we will manage its entire registration lifecycle, and
220  // retain full ownership over the object until we need to destroy it.
221  auto newHead = new AutowirableSlotFn<T, Fn>(std::shared_ptr<CoreContext>(), std::forward<Fn>(fn));
222 
223  // Append to our list in a lock-free way. This is a fairly standard way to do a lock-free append to
224  // a singly linked list; the only unusual aspect is the use of "this" as a tombstone indicator (IE,
225  // to signify that the list should no longer be used).
226  AutowirableSlot<T>* pFirstChild;
227  do {
228  // Obtain the current first child, and keep it here so we know what to exchange out with later:
229  pFirstChild = m_pFirstChild;
230 
231  // Determine if we've been tombstoned at this piont:
232  if(pFirstChild == this) {
233  // Trivially satisfy, and then return. This might look like a leak, but it's not, because we know
234  // that Finalize is going to destroy the object.
235  newHead->SatisfyAutowiring(this->m_ptr);
236  newHead->Finalize();
237  return;
238  }
239 
240  // Try to set the forward link to the current head, and then update our own flink;
241  newHead->SetFlink(pFirstChild);
242  } while(!m_pFirstChild.compare_exchange_weak(pFirstChild, newHead, std::memory_order_acquire));
243  }
244 
245  // Base overrides:
246  DeferrableAutowiring* ReleaseDependentChain(void) override {
247  // Rip the head off the list, replace it with a tombstone:
248  auto pFirstChild = m_pFirstChild.exchange(this, std::memory_order_relaxed);
249 
250  // If we got the tombstone back, we have nothing to return. Otherwise return the element.
251  return pFirstChild == this ? nullptr : pFirstChild;
252  }

285  // If you are getting an error tracked to this line, ensure that class T is totally
286  // defined at the point where the Autowired instance is constructed. Generally,
287  // such errors are tracked to missing header files. A common mistake, for instance,
288  // is to do something like this:
289  //
290  // class MyClass;
291  //
292  // struct MyStructure {
293  // Autowired<MyClass> m_member;
294  // };
295  //
296  // At the time m_member is instantiated, MyClass is an incomplete type. So, when the
297  // compiler tries to instantiate AutowiredCreator::Create (the function you're in right
298  // now!) it finds that it can't create a new instance of type MyClass because it has
299  // no idea how to construct it!
300  //
301  // This problem can be fixed two ways: You can include the definition of MyClass before
